### 2. ViewModel (1 час)

Создайте `EditorViewModel`:

```csharp
public partial class EditorViewModel : ObservableObject
{
    [ObservableProperty]
    private ObservableCollection<ShapeViewModel> _shapes;

    [ObservableProperty]
    private ShapeViewModel _selectedShape;

    [ObservableProperty]
    private double _zoom = 1.0;

    [ObservableProperty]
    private double _panOffsetX;

    [ObservableProperty]
    private double _panOffsetY;

    [ObservableProperty]
    private string _mousePosition;

    [ObservableProperty]
    private string _currentTool;

    public EditorViewModel()
    {
        Shapes = new ObservableCollection<ShapeViewModel>();
        
        ZoomInCommand = new RelayCommand(ZoomIn);
        ZoomOutCommand = new RelayCommand(ZoomOut);
        FitToScreenCommand = new RelayCommand(FitToScreen);
        ClearCommand = new RelayCommand(Clear);
        SelectLineToolCommand = new RelayCommand(() => CurrentTool = "Line");
        SelectRectangleToolCommand = new RelayCommand(() => CurrentTool = "Rectangle");
        SelectEllipseToolCommand = new RelayCommand(() => CurrentTool = "Ellipse");
    }

    [RelayCommand]
    private void ZoomIn() => Zoom = Math.Min(Zoom * 1.2, 10.0);

    [RelayCommand]
    private void ZoomOut() => Zoom = Math.Max(Zoom / 1.2, 0.1);

    [RelayCommand]
    private void FitToScreen()
    {
        Zoom = 1.0;
        PanOffsetX = 0;
        PanOffsetY = 0;
    }

    [RelayCommand]
    private void Clear() => Shapes.Clear();

    public void AddShape(double x1, double y1, double x2, double y2)
    {
        ShapeViewModel shape = CurrentTool switch
        {
            "Line" => new LineViewModel { StartX = x1, StartY = y1, EndX = x2, EndY = y2 },
            "Rectangle" => new RectangleViewModel 
            { 
                X = Math.Min(x1, x2), 
                Y = Math.Min(y1, y2),
                Width = Math.Abs(x2 - x1),
                Height = Math.Abs(y2 - y1)
            },
            "Ellipse" => new EllipseViewModel
            {
                X = Math.Min(x1, x2),
                Y = Math.Min(y1, y2),
                Width = Math.Abs(x2 - x1),
                Height = Math.Abs(y2 - y1)
            },
            _ => null
        };

        if (shape != null)
        {
            Shapes.Add(shape);
        }
    }
}
```

### 5. Рисование мышью (45 минут)

Реализуйте рисование в code-behind:

```csharp
public partial class MainWindow : Window
{
    private bool _isDrawing;
    private Point _startPoint;

    public MainWindow()
    {
        InitializeComponent();
        
        DrawingCanvas.MouseLeftButtonDown += DrawingCanvas_MouseLeftButtonDown;
        DrawingCanvas.MouseMove += DrawingCanvas_MouseMove;
        DrawingCanvas.MouseLeftButtonUp += DrawingCanvas_MouseLeftButtonUp;
    }

    private void DrawingCanvas_MouseLeftButtonDown(object sender, MouseButtonEventArgs e)
    {
        _isDrawing = true;
        _startPoint = e.GetPosition(DrawingCanvas);
        DrawingCanvas.CaptureMouse();
    }

    private void DrawingCanvas_MouseMove(object sender, MouseEventArgs e)
    {
        if (!_isDrawing) return;

        var vm = (EditorViewModel)DataContext;
        var currentPoint = e.GetPosition(DrawingCanvas);

        // Обновление позиции мыши в StatusBar
        vm.MousePosition = $"X: {currentPoint.X:F0}, Y: {currentPoint.Y:F0}";
    }

    private void DrawingCanvas_MouseLeftButtonUp(object sender, MouseButtonEventArgs e)
    {
        if (!_isDrawing) return;

        _isDrawing = false;
        DrawingCanvas.ReleaseMouseCapture();

        var vm = (EditorViewModel)DataContext;
        var endPoint = e.GetPosition(DrawingCanvas);

        // Учёт Zoom и Pan
        double zoom = vm.Zoom;
        double panX = vm.PanOffsetX;
        double panY = vm.PanOffsetY;

        var start = new Point(
            (_startPoint.X - panX) / zoom,
            (_startPoint.Y - panY) / zoom
        );
        var end = new Point(
            (endPoint.X - panX) / zoom,
            (endPoint.Y - panY) / zoom
        );

        vm.AddShape(start.X, start.Y, end.X, end.Y);
    }
}
```

### 6. Сетка через OnRender (45 минут)

Создайте кастомный Canvas для сетки:

```csharp
public class GridCanvas : Canvas
{
    public static readonly DependencyProperty GridStepProperty =
        DependencyProperty.Register(
            "GridStep",
            typeof(double),
            typeof(GridCanvas),
            new PropertyMetadata(20.0, (d, e) => d.InvalidateVisual())
        );

    public static readonly DependencyProperty ZoomProperty =
        DependencyProperty.Register(
            "Zoom",
            typeof(double),
            typeof(GridCanvas),
            new PropertyMetadata(1.0, (d, e) => d.InvalidateVisual())
        );

    public double GridStep
    {
        get => (double)GetValue(GridStepProperty);
        set => SetValue(GridStepProperty, value);
    }

    public double Zoom
    {
        get => (double)GetValue(ZoomProperty);
        set => SetValue(ZoomProperty, value);
    }

    protected override void OnRender(DrawingContext drawingContext)
    {
        base.OnRender(drawingContext);

        double step = GridStep * Zoom;
        double width = ActualWidth;
        double height = ActualHeight;

        // Рисуем вертикальные линии
        for (double x = 0; x < width; x += step)
        {
            drawingContext.DrawLine(
                new Pen(Brushes.LightGray, 0.5),
                new Point(x, 0),
                new Point(x, height)
            );
        }

        // Рисуем горизонтальные линии
        for (double y = 0; y < height; y += step)
        {
            drawingContext.DrawLine(
                new Pen(Brushes.LightGray, 0.5),
                new Point(0, y),
                new Point(width, y)
            );
        }
    }
}
```

```xml
<!-- Использование -->
<local:GridCanvas x:Name="GridCanvas"
                  GridStep="20"
                  Zoom="{Binding Zoom}"
                  Background="White"/>
```
