### 2. Async загрузка с Progress (45 минут)

Реализуйте асинхронную загрузку:

```csharp
public partial class MainViewModel : ObservableObject
{
    [ObservableProperty]
    private bool _isLoading;

    [ObservableProperty]
    private int _loadProgress;

    [ObservableProperty]
    private string _statusMessage;

    [ObservableProperty]
    private ObservableCollection<DataItem> _items;

    public ICommand LoadDataCommand { get; }
    public ICommand CancelLoadCommand { get; }

    private CancellationTokenSource _cancellationTokenSource;

    public MainViewModel()
    {
        Items = new ObservableCollection<DataItem>();
        LoadDataCommand = new AsyncRelayCommand(LoadDataAsync, CanLoadData);
        CancelLoadCommand = new RelayCommand(CancelLoad);
    }

    private bool CanLoadData() => !_isLoading;

    private async Task LoadDataAsync()
    {
        IsLoading = true;
        StatusMessage = "Загрузка данных...";
        _cancellationTokenSource = new CancellationTokenSource();

        try
        {
            var progress = new Progress<int>(p => LoadProgress = p);

            var items = await Task.Run(
                () => LoadDataFromDatabase(progress),
                _cancellationTokenSource.Token);

            Items = new ObservableCollection<DataItem>(items);
            StatusMessage = $"Загружено {Items.Count} элементов";
        }
        catch (OperationCanceledException)
        {
            StatusMessage = "Загрузка отменена";
        }
        catch (Exception ex)
        {
            StatusMessage = $"Ошибка: {ex.Message}";
        }
        finally
        {
            IsLoading = false;
            _cancellationTokenSource = null;
        }
    }

    private List<DataItem> LoadDataFromDatabase(IProgress<int> progress)
    {
        var items = new List<DataItem>();
        
        for (int i = 0; i < 10000; i++)
        {
            _cancellationTokenSource.Token.ThrowIfCancellationRequested();
            
            Thread.Sleep(10); // Имитация загрузки
            
            items.Add(new DataItem
            {
                Id = i,
                Title = $"Item {i}",
                Description = $"Description {i}"
            });
            
            if (i % 100 == 0)
            {
                progress.Report(i / 100);
            }
        }
        
        return items;
    }

    private void CancelLoad()
    {
        _cancellationTokenSource?.Cancel();
    }
}
```

### 3. Freezable Optimization (30 минут)

Оптимизируйте рисование:

```csharp
// OptimizedRenderer.cs
public class OptimizedRenderer
{
    private readonly SolidColorBrush _cachedBrush;
    private readonly Pen _cachedPen;
    
    public OptimizedRenderer()
    {
        // Создаём и замораживаем кисти
        _cachedBrush = new SolidColorBrush(Colors.Blue);
        _cachedBrush.Freeze();
        
        _cachedPen = new Pen(_cachedBrush, 2);
        _cachedPen.Freeze();
    }
    
    public void Render(DrawingContext context)
    {
        // Используем замороженные объекты
        context.DrawRectangle(_cachedBrush, _cachedPen, new Rect(0, 0, 100, 50));
    }
}
```

### 5. Performance Monitor (45 минут)

Создайте монитор производительности:

```csharp
// PerformanceMonitor.cs
public class PerformanceMonitor : INotifyPropertyChanged
{
    private readonly DispatcherTimer _timer;
    private int _frameCount;
    private double _fps;

    public double FPS
    {
        get => _fps;
        private set
        {
            _fps = value;
            PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(nameof(FPS)));
        }
    }

    public PerformanceMonitor()
    {
        _timer = new DispatcherTimer
        {
            Interval = TimeSpan.FromMilliseconds(500)
        };
        _timer.Tick += (s, e) =>
        {
            FPS = _frameCount * 2; // 500ms * 2 = 1 second
            _frameCount = 0;
        };
        _timer.Start();
        
        CompositionTarget.Rendering += (s, e) => _frameCount++;
    }

    public event PropertyChangedEventHandler PropertyChanged;
}
```

```xml
<!-- Отображение FPS -->
<StackPanel Orientation="Horizontal" HorizontalAlignment="Right">
    <TextBlock Text="FPS: " Foreground="White"/>
    <TextBlock Text="{Binding FPS, ElementName=perfMonitor}" 
               Foreground="LightGreen"
               FontWeight="Bold"/>
</StackPanel>
```

<Window x:Class="WpfApp.MainWindow">
    <Grid Background="{DynamicResource WindowBackgroundBrush}">
        <Grid.RowDefinitions>
            <RowDefinition Height="Auto"/>
            <RowDefinition Height="*"/>
            <RowDefinition Height="Auto"/>
        </Grid.RowDefinitions>
        
        <!-- Toolbar -->
        <ToolBar Grid.Row="0">
            <Button Content="Загрузить данные" 
                    Command="{Binding LoadDataCommand}"
                    IsEnabled="{Binding IsLoading, Converter={StaticResource InverseBool}}"/>
            <Button Content="Отменить" 
                    Command="{Binding CancelLoadCommand}"
                    IsEnabled="{Binding IsLoading}"/>
        </ToolBar>
        
        <!-- Progress -->
        <Grid Grid.Row="1" Visibility="{Binding IsLoading, Converter={StaticResource BoolToVisibility}}">
            <StackPanel VerticalAlignment="Center" HorizontalAlignment="Center">
                <ProgressBar Width="300" Height="20" 
                             Value="{Binding LoadProgress}"
                             Minimum="0" Maximum="100"/>
                <TextBlock Text="{Binding StatusMessage}" 
                           HorizontalAlignment="Center"
                           Margin="0,10,0,0"/>
            </StackPanel>
        </Grid>
        
        <!-- ListBox с виртуализацией -->
        <ListBox Grid.Row="1"
                 ItemsSource="{Binding Items}"
                 Visibility="{Binding IsLoading, Converter={StaticResource InverseBoolToVisibility}}">
            <ListBox.ItemsPanel>
                <ItemsPanelTemplate>
                    <VirtualizingStackPanel/>
                </ItemsPanelTemplate>
            </ListBox.ItemsPanel>
        </ListBox>
