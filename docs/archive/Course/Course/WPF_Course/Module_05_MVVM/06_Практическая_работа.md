### 4. Dependency Injection (30 минут)

Настройте DI в App.xaml.cs:

```csharp
public partial class App : Application
{
    private readonly IHost _host;

    public App()
    {
        _host = Host.CreateDefaultBuilder()
            .ConfigureServices((context, services) =>
            {
                // Сервисы
                services.AddSingleton<ITaskService, TaskService>();
                services.AddSingleton<IDialogService, DialogService>();
                
                // ViewModel
                services.AddSingleton<MainViewModel>();
                services.AddTransient<TaskEditViewModel>();
            })
            .Build();
    }

    protected override void OnStartup(StartupEventArgs e)
    {
        base.OnStartup(e);
        
        var mainWindow = new MainWindow
        {
            DataContext = _host.Services.GetRequiredService<MainViewModel>()
        };
        mainWindow.Show();
    }
}
```

### 7. Конвертеры (15 минут)

Создайте конвертеры:

```csharp
// BoolToUnderlineConverter
public class BoolToUnderlineConverter : IValueConverter
{
    public object Convert(object value, Type targetType, 
                          object parameter, CultureInfo culture)
    {
        return (bool)value ? TextDecorations.Strikethrough : null;
    }

    public object ConvertBack(...) => throw new NotImplementedException();
}

// PriorityToColorConverter
public class PriorityToColorConverter : IValueConverter
{
    public object Convert(object value, Type targetType, 
                          object parameter, CultureInfo culture)
    {
        int priority = (int)value;
        return priority switch
        {
            1 => Brushes.Green,
            2 => Brushes.LightGreen,
            3 => Brushes.Yellow,
            4 => Brushes.Orange,
            5 => Brushes.Red,
            _ => Brushes.Black
        };
    }

    public object ConvertBack(...) => throw new NotImplementedException();
}
```
