## Типичные ошибки студентов

### ❌ Неправильно

```xml
<!-- 1. Grid вместо Canvas для абсолютного позиционирования -->
<Grid>
    <Button Margin="100,50" Content="Click"/>
</Grid>

<!-- 2. Обычные свойства вместо DP -->
public string Text { get; set; } // ❌

<!-- 3. Code-behind вместо MVVM -->
private void Button_Click(object sender, RoutedEventArgs e)
{
    // Бизнес-логика в UI слое ❌
}

<!-- 4. StaticResource для тем -->
<Button Background="{StaticResource ThemeBrush}"/> 
// Не обновится при смене темы ❌
```

### ✅ Правильно

```xml
<!-- 1. Canvas для абсолютного позиционирования -->
<Canvas>
    <Button Canvas.Left="100" Canvas.Top="50" Content="Click"/>
</Canvas>

<!-- 2. Dependency Property -->
public string Text
{
    get => (string)GetValue(TextProperty);
    set => SetValue(TextProperty, value);
}

<!-- 3. MVVM с Commands -->
public ICommand ClickCommand { get; }
// Команда в ViewModel ✅

<!-- 4. DynamicResource для тем -->
<Button Background="{DynamicResource ThemeBrush}"/>
// Обновится при смене темы ✅
```
